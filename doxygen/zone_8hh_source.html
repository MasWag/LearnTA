182  [[nodiscard]] std::vector<double> sample() const {
183  std::vector<double> valuation;
184  std::size_t N = this->getNumOfVar();
185  valuation.resize(N);
186  for (int i = 0; i < N; i++) {
187  Bounds lowerBound = this->value(0, i + 1);
188  Bounds upperBound = this->value(i + 1, 0);
189  if (isPoint(upperBound, lowerBound)) {
190  valuation[i] = upperBound.first;
191  } else {
192  double lower = -lowerBound.first;
193  double upper = upperBound.first;
194  for (int j = 0; j < i; j++) {
195  Bounds tmpLowerBound = this->value(j + 1, i + 1);
196  Bounds tmpUpperBound = this->value(i + 1, j + 1);
197  lower = std::max(lower, -tmpLowerBound.first - valuation[j]);
198  upper = std::min(upper, tmpUpperBound.first + valuation[j]);
199  }
200  assert(lower <= upper);
201  if (lowerBound.second) {
202  valuation[i] = lower;
203  } else if (upper - lower > 0.5) {
204  valuation[i] = lower + 0.5;
205  } else {
206  valuation[i] = (lower + upper) * 0.5;
207  }
208  }
209  }
