170  [[nodiscard]] std::vector<double> sample() {
171  this->canonize();
172  assert(this->isSatisfiableNoCanonize());
173  std::vector<double> valuation;
174  std::size_t N = this->getNumOfVar();
175  valuation.resize(N);
176  for (int i = 0; i < N; i++) {
177  Bounds lowerBound = this->value(0, i + 1);
178  Bounds upperBound = this->value(i + 1, 0);
179  if (isPoint(upperBound, lowerBound)) {
180  valuation[i] = upperBound.first;
181  } else {
182  double lower = -lowerBound.first;
183  double upper = upperBound.first;
184  for (int j = 0; j < i; j++) {
185  Bounds tmpLowerBound = this->value(j + 1, i + 1);
186  Bounds tmpUpperBound = this->value(i + 1, j + 1);
187  lower = std::max(lower, -tmpLowerBound.first + valuation[j]);
188  upper = std::min(upper, tmpUpperBound.first + valuation[j]);
189  }
190  assert(lower <= upper);
191  if (lowerBound.second) {
192  valuation[i] = lower;
193  } else if (upper - lower > 0.5) {
194  valuation[i] = lower + 0.5;
195  } else {
196  valuation[i] = (lower + upper) * 0.5;
197  }
198  }
199  }
