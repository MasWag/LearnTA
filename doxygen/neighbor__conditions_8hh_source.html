78  static auto makeNeighbors(const ForwardRegionalElementaryLanguage &original,
79  const std::unordered_set<ClockVariables> &preciseClocks) {
80  const auto clockSize = original.getTimedCondition().size();
81  std::vector<ForwardRegionalElementaryLanguage> neighbors;
82 
83  // Construct the neighbors
84  auto neighborsCondition = TimedCondition(Zone::top(clockSize + 1));
85  for (std::size_t i = 0; i < clockSize; ++i) {
86  // The constraints of the form x \in I is used
87  neighborsCondition.restrictLowerBound(i, clockSize - 1,
88  original.getTimedCondition().getLowerBound(i, clockSize - 1));
89  neighborsCondition.restrictUpperBound(i, clockSize - 1,
90  original.getTimedCondition().getUpperBound(i, clockSize - 1));
91  for (std::size_t j = i + 1; j < clockSize; ++j) {
92  if ((preciseClocks.find(i) == preciseClocks.end()) ==
93  (preciseClocks.find(j) == preciseClocks.end())) {
94  // The constraints of the form x - y \in I is used if both are precise or imprecise
95  neighborsCondition.restrictLowerBound(i, j - 1, original.getTimedCondition().getLowerBound(i, j - 1));
96  neighborsCondition.restrictUpperBound(i, j - 1, original.getTimedCondition().getUpperBound(i, j - 1));
97  } else {
98  neighborsCondition.restrictLowerBound(i, j - 1, Bounds{0, true});
99  }
100  }
101  }
102  const auto simpleNeighborsCondition = neighborsCondition.enumerate();
103  neighbors.reserve(simpleNeighborsCondition.size());
104  std::transform(simpleNeighborsCondition.begin(), simpleNeighborsCondition.end(), std::back_inserter(neighbors),
105  [&](const auto &neighborCondition) {
106  return ForwardRegionalElementaryLanguage::fromTimedWord(
107  ElementaryLanguage{original.getWord(), neighborCondition}.sample());
108  });
109 
110  return neighbors;
111  }
