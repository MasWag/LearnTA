36  this->zone = Zone::top(accumulatedDuration.size() + 1);
37  for (int i = 0; i < accumulatedDuration.size(); ++i) {

39  // T_{i, j} = accumulatedDuration.at(i) - accumulatedDuration.at(j + 1)
40  const auto concreteDifference = accumulatedDuration.at(i) -

132  const size_t M = another.size();

137  for (int i = N + 1; i < N + M; ++i) {
138  // Copy \f$A_{(1, 0), (N, 1)}\f$ to \f$C_{(1, i), (N, 1)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.

140  // Copy \f$A_{(0, 1), (1, N)}\f$ to \f$C_{(i, 1), (1, N)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.
141  result.zone.value.block(i, 1, 1, N) = this->zone.value.block(0, 1, 1, N);
142  }
143  if (M >= 2) {

145  result.zone.value.block(N + 1, N + 1, M - 1, M - 1) = another.zone.value.block(2, 2, M - 1, M - 1);
146  // Copy \f$B_{(2, 0), (M - 1, 1)}\f$ to \f$C_{(N + 1, 0), (M - 1, 1)}\f$.

148  // Copy \f$B_{(0, 2), (1, M - 1)}\f$ to \f$C_{(0, N + 1), (1, M - 1)}\f$.
149  result.zone.value.block(0, N + 1, 1, M - 1) = another.zone.value.block(0, 2, 1, M - 1);
150  }

153  result.zone.value.block(N + 1, i, M - 1, 1).array() += another.zone.value.block(2, 1, M - 1, 1).array();
154  // Add \f$B_{(1, 2), (1, M - 1)}\f$ to \f$C_{(i, N + 1), (1, M - 1)}\f$ for each \f$i \in \{1, \dots, N\}\f$.
155  result.zone.value.block(i, N + 1, 1, M - 1).array() += another.zone.value.block(1, 2, 1, M - 1).array();

279  void enumerate(std::vector<TimedCondition> &simpleConditions) const {
280  if (this->isSimple()) {
281  simpleConditions = {*this};
282  return;
283  }
284  std::vector<TimedCondition> currentConditions = {*this};
285  for (int i = 0; i < this->size(); i++) {
286  for (int j = i; j < this->size(); j++) {
287  std::vector<TimedCondition> nextConditions;
288  for (const TimedCondition &timedCondition: currentConditions) {
289  if (timedCondition.isSimple()) {
290  simpleConditions.push_back(timedCondition);
291  continue;
292  }
293  auto lowerBound = timedCondition.getLowerBound(i, j);
294  const auto upperBound = timedCondition.getUpperBound(i, j);
295  if (isPoint(upperBound, lowerBound) or isUnitOpen(upperBound, lowerBound)) {
296  nextConditions.push_back(timedCondition);
297  } else {
298  while (-lowerBound <= upperBound) {
299  auto currentTimedCondition = timedCondition;
300  if (lowerBound.second) {
301  currentTimedCondition.restrictLowerBound(i, j, lowerBound);
302  currentTimedCondition.restrictUpperBound(i, j, -lowerBound);
303  lowerBound.second = false;
304  } else {
305  currentTimedCondition.restrictLowerBound(i, j, lowerBound);
306  currentTimedCondition.restrictUpperBound(i, j, {-lowerBound.first + 1, false});
307  lowerBound = {lowerBound.first - 1, true};
308  }
309  if (currentTimedCondition.isSimple()) {
310  simpleConditions.push_back(currentTimedCondition);
311  } else {
312  nextConditions.push_back(currentTimedCondition);
313  }
314  }
315  }
316  }
317  currentConditions = std::move(nextConditions);
318  if (currentConditions.empty()) {
319  return;
320  }
321  }
