38  for (std::size_t i = 0; i < accumulatedDuration.size(); ++i) {

42  ((j + 1 < accumulatedDuration.size()) ? accumulatedDuration.at(j + 1) : 0);

151  [[nodiscard]] TimedCondition operator+(const TimedCondition &another) const {
152  const size_t N = this->size();
153  const size_t M = another.size();
154  Zone result = Zone::top(N + M);
155  // Copy \f$A_{(0, 0), (N + 1, N + 1)}\f$ to \f$C_{(0, 0), (N + 1, N + 1)}\f$.
156  result.value.block(0, 0, N + 1, N + 1) = this->zone.value;
157  for (std::size_t i = N + 1; i < N + M; ++i) {
158  // Copy \f$A_{(1, 0), (N, 1)}\f$ to \f$C_{(1, i), (N, 1)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.
159  result.value.block(1, i, N, 1) = this->zone.value.block(1, 0, N, 1);
160  // Copy \f$A_{(0, 1), (1, N)}\f$ to \f$C_{(i, 1), (1, N)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.
161  result.value.block(i, 1, 1, N) = this->zone.value.block(0, 1, 1, N);
162  }
163  if (M >= 2) {
164  // Copy \f$B_{(2, 2), (M - 1, M - 1)}\f$ to \f$C_{(N + 1, N + 1), (M - 1, M - 1)}\f$.
165  result.value.block(N + 1, N + 1, M - 1, M - 1) = another.zone.value.block(2, 2, M - 1, M - 1);
166  // Copy \f$B_{(2, 0), (M - 1, 1)}\f$ to \f$C_{(N + 1, 0), (M - 1, 1)}\f$.
167  result.value.block(N + 1, 0, M - 1, 1) = another.zone.value.block(2, 0, M - 1, 1);
168  // Copy \f$B_{(0, 2), (1, M - 1)}\f$ to \f$C_{(0, N + 1), (1, M - 1)}\f$.
169  result.value.block(0, N + 1, 1, M - 1) = another.zone.value.block(0, 2, 1, M - 1);
170  }
171  for (std::size_t i = 1; i <= N; ++i) {
172  // Add \f$B_{(2, 1), (M - 1, 1)}\f$ to \f$C_{(N + 1, i), (M - 1, 1)}\f$ for each \f$i \in \{1, \dots, N\}\f$.
173  result.value.block(N + 1, i, M - 1, 1).array() += another.zone.value.block(2, 1, M - 1, 1).array();
174  // Add \f$B_{(1, 2), (1, M - 1)}\f$ to \f$C_{(i, N + 1), (1, M - 1)}\f$ for each \f$i \in \{1, \dots, N\}\f$.
175  result.value.block(i, N + 1, 1, M - 1).array() += another.zone.value.block(1, 2, 1, M - 1).array();
176  }
177  // Add \f$B_{(1, 0)}\f$ to \f$C_{(1, 0), (N, 1)}\f$.
178  result.value.block(1, 0, N, 1).array() += another.zone.value(1, 0);
179  // Add \f$B_{(0, 1)}\f$ to \f$C_{(0, 1), (1, N)}\f$.
180  result.value.block(0, 1, 1, N).array() += another.zone.value(0, 1);
181 
182  result.canonize();
183 
184  return TimedCondition{std::move(result)};
185  }

297  void enumerate(std::vector<TimedCondition> &simpleConditions) const {
298  if (this->isSimple()) {
299  simpleConditions = {*this};
300  return;
301  }
302  std::vector<TimedCondition> currentConditions = {*this};
303  for (std::size_t i = 0; i < this->size(); i++) {
304  for (std::size_t j = i; j < this->size(); j++) {
305  std::vector<TimedCondition> nextConditions;
306  for (const TimedCondition &timedCondition: currentConditions) {
307  if (timedCondition.isSimple()) {
308  simpleConditions.push_back(timedCondition);
309  continue;
310  }
311  auto lowerBound = timedCondition.getLowerBound(i, j);
312  const auto upperBound = timedCondition.getUpperBound(i, j);
313  if (isPoint(upperBound, lowerBound) or isUnitOpen(upperBound, lowerBound)) {
314  nextConditions.push_back(timedCondition);
315  } else {
316  Bounds currentUpperBound = lowerBound.second ? -lowerBound : std::make_pair(-lowerBound.first + 1, false);
317  while (currentUpperBound <= upperBound) {
318  auto currentTimedCondition = timedCondition;
319  currentTimedCondition.restrictLowerBound(i, j, lowerBound, false);
320  currentTimedCondition.restrictUpperBound(i, j, currentUpperBound, false);
321  if (lowerBound.second) {
322  currentUpperBound = {-lowerBound.first + 1, false};
323  lowerBound.second = false;
324  } else {
325  currentUpperBound = std::make_pair(-lowerBound.first + 1, true);
326  lowerBound = {lowerBound.first - 1, true};
327  }
328  if (currentTimedCondition.isSimple()) {
329  simpleConditions.push_back(currentTimedCondition);
330  } else {
331  nextConditions.push_back(currentTimedCondition);
332  }
333  }
334  }
335  }
336  currentConditions = std::move(nextConditions);
337  if (currentConditions.empty()) {
338  return;
339  }
340  }
