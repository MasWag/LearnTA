167  std::vector<TimedCondition> nextConditions;
168  for (const TimedCondition &timedCondition: currentConditions) {
169  if (timedCondition.isSimple()) {
170  simpleConditions.push_back(timedCondition);
171  continue;
172  }
173  auto lowerBound = timedCondition.getLowerBound(i, j);
174  const auto upperBound = timedCondition.getUpperBound(i, j);
175  if (isPoint(upperBound, lowerBound) or isUnitOpen(upperBound, lowerBound)) {
176  nextConditions.push_back(timedCondition);
177  } else {
178  while (-lowerBound <= upperBound) {
179  auto currentTimedCondition = timedCondition;
180  if (lowerBound.second) {
181  currentTimedCondition.restrictLowerBound(i, j, lowerBound);
182  currentTimedCondition.restrictUpperBound(i, j, -lowerBound);
