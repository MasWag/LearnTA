107  TimedCondition result;
108  const size_t N = this->size();
109  const size_t M = another.size();
110  result = TimedCondition();
111  result.zone = Zone::top(N + M);
112  // Copy \f$A_{(0, 0), (N + 1, N + 1)}\f$ to \f$C_{(0, 0), (N + 1, N + 1)}\f$.
113  result.zone.value.block(0, 0, N + 1, N + 1) = this->zone.value;
114  for (int i = N + 1; i < N + M; ++i) {
115  // Copy \f$A_{(1, 0), (N, 1)}\f$ to \f$C_{(1, i), (N, 1)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.

117  // Copy \f$A_{(0, 1), (1, N)}\f$ to \f$C_{(i, 1), (1, N)}\f$ for each \f$i \in \{N, N + 1, \dots, N + M - 1\}\f$.
118  result.zone.value.block(i, 1, 1, N) = this->zone.value.block(0, 1, 1, N);
119  }
120  if (M >= 2) {
121  // Copy \f$B_{(2, 2), (M - 1, M - 1)}\f$ to \f$C_{(N + 1, N + 1), (M - 1, M - 1)}\f$.
122  result.zone.value.block(N + 1, N + 1, M - 1, M - 1) = another.zone.value.block(2, 2, M - 1, M - 1);
123  // Copy \f$B_{(2, 0), (M - 1, 1)}\f$ to \f$C_{(N + 1, 0), (M - 1, 1)}\f$.
124  result.zone.value.block(N + 1, 0, M - 1, 1) = another.zone.value.block(2, 0, M - 1, 1);
125  // Copy \f$B_{(0, 2), (1, M - 1)}\f$ to \f$C_{(0, N + 1), (1, M - 1)}\f$.
126  result.zone.value.block(0, N + 1, 1, M - 1) = another.zone.value.block(0, 2, 1, M - 1);
127  }
128  for (int i = 1; i <= N; ++i) {
129  // Add \f$B_{(2, 1), (M - 1, 1)}\f$ to \f$C_{(N + 1, i), (M - 1, 1)}\f$ for each \f$i \in \{1, \dots, N\}\f$.
130  result.zone.value.block(N + 1, i, M - 1, 1).array() += another.zone.value.block(2, 1, M - 1, 1).array();
131  // Add \f$B_{(1, 2), (1, M - 1)}\f$ to \f$C_{(i, N + 1), (1, M - 1)}\f$ for each \f$i \in \{1, \dots, N\}\f$.
132  result.zone.value.block(i, N + 1, 1, M - 1).array() += another.zone.value.block(1, 2, 1, M - 1).array();

137  result.zone.value.block(0, 1, 1, N).array() += another.zone.value(0, 1);

139  result.zone.canonize();

256  void enumerate(std::vector<TimedCondition> &simpleConditions) const {
257  if (this->isSimple()) {
258  simpleConditions = {*this};
259  return;
260  }
261  std::vector<TimedCondition> currentConditions = {*this};
262  for (int i = 0; i < this->size(); i++) {
263  for (int j = i; j < this->size(); j++) {
264  std::vector<TimedCondition> nextConditions;
265  for (const TimedCondition &timedCondition: currentConditions) {
266  if (timedCondition.isSimple()) {
267  simpleConditions.push_back(timedCondition);
268  continue;
269  }
270  auto lowerBound = timedCondition.getLowerBound(i, j);
271  const auto upperBound = timedCondition.getUpperBound(i, j);
272  if (isPoint(upperBound, lowerBound) or isUnitOpen(upperBound, lowerBound)) {
273  nextConditions.push_back(timedCondition);
274  } else {
275  while (-lowerBound <= upperBound) {
276  auto currentTimedCondition = timedCondition;
277  if (lowerBound.second) {
278  currentTimedCondition.restrictLowerBound(i, j, lowerBound);
279  currentTimedCondition.restrictUpperBound(i, j, -lowerBound);
280  lowerBound.second = false;
281  } else {
282  currentTimedCondition.restrictLowerBound(i, j, lowerBound);
283  currentTimedCondition.restrictUpperBound(i, j, {-lowerBound.first + 1, false});
284  lowerBound = {lowerBound.first - 1, true};
285  }
286  if (currentTimedCondition.isSimple()) {
287  simpleConditions.push_back(currentTimedCondition);
288  } else {
289  nextConditions.push_back(currentTimedCondition);
290  }
291  }
292  }
293  }
294  currentConditions = std::move(nextConditions);
295  if (currentConditions.empty()) {
296  return;
297  }
298  }
