94  void moveToP(const std::size_t index) {
95  // index should not be in P yet
96  assert(pIndices.find(index) == pIndices.end());
97  // The index should be in a valid range
98  assert(index < prefixes.size());
99  pIndices.insert(index);
100  // Add successors to the prefixes
101  prefixes.reserve(prefixes.size() + alphabet.size() + 1);
102  for (Alphabet c: alphabet) {
103  discreteSuccessors[std::make_pair(index, c)] = prefixes.size();
104  prefixes.emplace_back(prefixes.at(index).successor(c));
105  }
106  continuousSuccessors[index] = prefixes.size();
107  prefixes.emplace_back(prefixes.at(index).successor());
108  // fill the observation table
109  refreshTable();
110  // index should point P
111  assert(pIndices.find(index) != pIndices.end());
112  // the discrete successors of prefixes.at(index) should be in ext(P)
113  assert(std::all_of(alphabet.begin(), alphabet.end(), [&](Alphabet c) {
114  return 0 <= discreteSuccessors.at(std::make_pair(index, c)) &&
115  discreteSuccessors.at(std::make_pair(index, c)) < prefixes.size();
116  }));
117  assert(std::all_of(alphabet.begin(), alphabet.end(), [&](Alphabet c) {
118  return pIndices.find(discreteSuccessors.at(std::make_pair(index, c))) == pIndices.end();
119  }));
120  assert(pIndices.find(index) != pIndices.end());
121  // the continuous successor of prefixes.at(index) should be in ext(P)
122  assert(continuousSuccessors.at(index) < prefixes.size());
123  assert(pIndices.find(continuousSuccessors.at(index)) == pIndices.end());
124  // We add continuous successors to P if it is clearly not saturated
125  if (!equivalentWithMemo(continuousSuccessors.at(index), index)) {

318  bool close() {
319  for (std::size_t i = 0; i < this->prefixes.size(); i++) {
320  // Skip if this prefix is in P
321  if (this->inP(i)) {
322  continue;
323  }
324  auto prefix = this->prefixes.at(i);
325  auto prefixRow = this->table.at(i);
326 
327  // Use the memoized information for efficiency
328  bool found = false;
329  {
330  auto it = this->closedRelation.find(i);
331  if (it != closedRelation.end()) {
332  // When we already know that this prefix is equivalent to one of p \in P, we just confirm it.
333  for (auto targetIt = it->second.begin(); targetIt != it->second.end();) {
334  auto renamingRelation = targetIt->second;
335  if (equivalence(prefix, prefixRow, this->concatenations.at(i),
336  this->prefixes.at(targetIt->first), this->table.at(targetIt->first),
337  this->concatenations.at(targetIt->first),
338  this->suffixes, renamingRelation)) {
339  if (this->inP(targetIt->first)) {
340  found = true;
341  break;
342  } else {
343  ++targetIt;
344  }
345  } else {
346  targetIt = it->second.erase(targetIt);
347  }
348  }
349  }
350  }
351 
352  // We have no idea of the target prefix, and we find an equivalent row
353  if (!found) {
354  // First, we try to "jump" to the same state
355  found = std::any_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto j) {
356  return this->continuousSuccessors.at(j) == i && equivalentWithMemo(i, j);
357  });
358  if (!found) {
359  found = std::any_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto j) {
360  return equivalentWithMemo(i, j);
361  });
362  }
363  }
364  if (!found) {
365  // The observation table is not closed
366 #ifndef NDEBUG
367  const auto prePSize = this->pIndices.size();
368 #endif
369  this->moveToP(i);
370  assert(prePSize < this->pIndices.size());
371  LOG_REFINEMENT_INFO << "Observation table is not closed because of " << this->prefixes.at(i);
372  this->refreshTable();
373  return false;
374  }
375  }
376 
377  // Assert that we have prefixes for any successors
378  assert(std::all_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto &pIndex) {
379  const auto successor = this->continuousSuccessors.at(pIndex);
380  return successor < this->prefixes.size();
381  }));
382  assert(std::all_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto &pIndex) {
383  return std::all_of(this->alphabet.begin(), this->alphabet.end(), [&](const auto &action) {
384  const auto successor = this->discreteSuccessors.at(std::make_pair(pIndex, action));
385  return successor < this->prefixes.size();
386  });
387  }));
388  // Assert that we have the information of the closedness in closed relation
389  for (std::size_t pIndex = 0; pIndex < this->prefixes.size(); ++pIndex) {
390  assert(this->inP(pIndex) || std::any_of(this->closedRelation.at(pIndex).begin(),
391  this->closedRelation.at(pIndex).end(),
392  [&](const auto &rPair) {
393  return this->inP(rPair.first);
394  }));
395  }
396  assert(std::all_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto &pIndex) {
397  return this->pIndices.find(this->continuousSuccessors.at(pIndex)) != this->pIndices.end() ||
398  std::any_of(this->closedRelation.at(this->continuousSuccessors.at(pIndex)).begin(),
399  this->closedRelation.at(this->continuousSuccessors.at(pIndex)).end(),
400  [&](const auto &rPair) {
401  return this->inP(rPair.first);
402  });
403  }));
404  assert(std::all_of(this->pIndices.begin(), this->pIndices.end(), [&](const auto &pIndex) {
405  return std::all_of(this->alphabet.begin(), this->alphabet.end(), [&](const auto &action) {
406  const auto successor = this->discreteSuccessors.at(std::make_pair(pIndex, action));
407  return this->inP(successor) || std::any_of(this->closedRelation.at(successor).begin(),
408  this->closedRelation.at(successor).end(),
409  [&](const std::pair<std::size_t, RenamingRelation> &rPair) {
410  return this->inP(rPair.first);
411  });
412  });
413  }));
414  return true;

652  std::stack<std::size_t> toSearch;
653  toSearch.push(rootIndex);
654  while(!toSearch.empty()) {
655  const auto currentIndex = toSearch.top();
656  toSearch.pop();
657  if (this->inP(currentIndex)) {
658  // Add the successors
659  toSearch.push(this->continuousSuccessors.at(currentIndex));
660  for (const auto action: this->alphabet) {
661  toSearch.push(this->discreteSuccessors.at(std::make_pair(currentIndex, action)));
662  }
663  } else {
664  const auto currentLanguage = this->prefixes.at(currentIndex);
665  const auto& currentCondition = currentLanguage.getTimedCondition();
666  for (const auto &[indexAfterMap, renaming]: this->closedRelation.at(currentIndex)) {
667  if (this->inP(indexAfterMap) && this->equivalentWithMemo(currentIndex, indexAfterMap)) {
668  // confirm that imprecise variables are not used
669  for (const ClockVariables clock: impreciseAfterMorphism) {
670  if (!currentCondition.isPoint(clock) &&
671  std::any_of(renaming.begin(), renaming.end(), is_second<std::size_t, std::size_t>(clock))) {
672  // clock is imprecise but used!!
673  LOG_REFINEMENT_INFO << "Observation table is renaming inconsistent: x"
674  << static_cast<int>(clock);
675  // Try to extend the suffixes
676  const TimedWord extension = currentLanguage.suffix(rootLanguage).sample();
677  for (const auto &suffix: this->suffixes) {
678  const auto newSuffix = BackwardRegionalElementaryLanguage::fromTimedWord(extension + suffix.sample());
679  if (!equivalent(i, rootIndex, newSuffix, rootRenaming) ||
680  !equivalent(currentIndex, indexAfterMap, newSuffix, renaming)) {
681  LOG_REFINEMENT_INFO << "renameConsistent: New suffix " << newSuffix << " is added";
682  this->suffixes.emplace_back(newSuffix);
683  this->refreshTable();
684  return false;
685  }
686  for (const auto &simple: (currentLanguage + suffix).enumerate()) {
687  const auto newSuffix2 = ForwardRegionalElementaryLanguage::fromTimedWord(simple.sample()).suffix(rootLanguage);
688  if (!equivalent(i, rootIndex, newSuffix2, rootRenaming) ||
689  !equivalent(currentIndex, indexAfterMap, newSuffix2, renaming)) {
690  LOG_REFINEMENT_INFO << "renameConsistent: New suffix " << newSuffix2

693  this->refreshTable();
694  return false;

697  }
698  LOG_REFINEMENT_INFO << "Failed to resolve the renaming inconsistency!!";
699  }
700  }
701  }
702  }
703  }
704  }
705  } else {
706  it = mapping.erase(it);
707  }
708  }
709  }
710  }
