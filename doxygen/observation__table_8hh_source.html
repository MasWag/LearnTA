51  void moveToP(std::size_t index) {
52  // index should not be in P yet
53  assert(pIndices.find(index) == pIndices.end());
54  pIndices.insert(index);
55  auto it = closedRelation.find(index);
56  if (it != closedRelation.end()) {
57  closedRelation.erase(it);
58  }
59  // Add successors to the prefixes
60  prefixes.reserve(prefixes.size() + alphabet.size() + 1);
61  for (Alphabet c: alphabet) {
62  prefixes.push_back(prefixes.at(index).successor(c));
63  discreteSuccessors.at(std::make_pair(index, c)) = prefixes.size();
64  }
65  prefixes.push_back(prefixes.at(index).successor());
66  continuousSuccessors.at(index) = prefixes.size();
67  // fill the observation table
68  refreshTable();
