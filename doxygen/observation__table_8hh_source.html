169  bool close() {

171  {
172  // Check if this prefix is in P
173  auto it = this->pIndices.find(i);
174  if (it != this->pIndices.end()) {
175  continue;
176  }
177  }
178  auto prefix = this->prefixes.at(i);
179  auto prefixRow = this->table.at(i);
180  bool found = false;
181  {
182  auto it = this->closedRelation.find(i);
183  if (it != closedRelation.end()) {
184  // When we already know that this prefix is equivalent to one of p \in P, we just confirm it.
185  for (auto targetIt = it->second.begin(); targetIt != it->second.end();) {
186  auto renamingRelation = targetIt->second;
187  if (equivalence(prefix, prefixRow,
188  this->prefixes.at(targetIt->first), this->table.at(targetIt->first), this->suffixes,
189  renamingRelation)) {
190  found = true;
191  break;
192  } else {
193  targetIt = it->second.erase(targetIt);
194  }
195  }
196  }
197  }
198  // We have no idea of the target prefix, and we find an equivalent row
199  if (!found) {

201  return equivalent(i, j);

224  bool consistent() {
225  for (const auto i: pIndices) {
226  for (const auto j: pIndices) {
227  if (i <= j) {
228  continue;
229  }
230  if (this->equivalentWithMemo(i, j)) {
231  // confirm the consistency
232  for (const auto action: this->alphabet) {
233  if (!this->equivalentWithMemo(this->discreteSuccessors.at({i, action}),
234  this->discreteSuccessors.at({j, action}))) {
235  BOOST_LOG_TRIVIAL(debug) << "Observation table is inconsistent because of discrete successor of "
236  << this->prefixes.at(i) << " and " << this->prefixes.at(j)
237  << " with action " << action;
238 
239  // The observation table is inconsistent due to a discrete successor
240  auto it = std::find_if(suffixes.begin(), suffixes.end(), [&](const auto &suffix) {
241  return !equivalent(i, j, suffix.predecessor(action));
242  });
243  // we assume that we have such a suffix
244  const auto newSuffix = it->predecessor(action);
245  suffixes.push_back(newSuffix);
246  BOOST_LOG_TRIVIAL(debug) << "New suffix " << newSuffix << " is added";
247 
248  this->refreshTable();
249  // i and j should not be equivalent after adding the new suffix
250  assert(!this->equivalentWithMemo(i, j));
251 
252  return false;
253  }
254  }
255  if (!this->equivalentWithMemo(this->continuousSuccessors.at(i), this->continuousSuccessors.at(j))) {
256  // The observation table is inconsistent due to a continuous successor
257  auto it = std::find_if(suffixes.begin(), suffixes.end(), [&](const auto &suffix) {
258  return !equivalent(i, j, suffix.predecessor());
259  });
260  // we assume that we have such a suffix
261  if (it != suffixes.end()) {
262  suffixes.push_back(it->predecessor());
263  } else {
264  // We find a set of suffixes to add
265  std::list<BackwardRegionalElementaryLanguage> allPredecessors;
266  allPredecessors.resize(suffixes.size());
267  std::transform(suffixes.begin(), suffixes.end(), allPredecessors.begin(), [](const auto &suffix) {
268  return suffix.predecessor();
269  });
270  if (equivalent(i, j, allPredecessors)) {
271  BOOST_LOG_TRIVIAL(info) << "Finding longer predecessors. This is slow";
272  auto preAllPredecessors = allPredecessors;
273  while (equivalent(i, j, allPredecessors)) {
274  std::list<BackwardRegionalElementaryLanguage> newAllPredecessors;
275  for (const auto &suffix: preAllPredecessors) {
276  try {
277  newAllPredecessors.push_back(suffix.predecessor());
278  } catch (...) {
279  }
280  }
281  if (newAllPredecessors.empty()) {
282  BOOST_LOG_TRIVIAL(fatal) << "Something is wrong in resolving continuous inconsistency";
283  abort();
284  }
285  preAllPredecessors = newAllPredecessors;
286  std::copy(preAllPredecessors.begin(), preAllPredecessors.end(),
287  std::back_inserter(allPredecessors));
288  }
289  }
290  auto jt = std::find_if_not(allPredecessors.begin(), allPredecessors.end(), [&](const auto &suffix) {
291  auto examinedPredecessor = allPredecessors;
292  auto kt = std::find(examinedPredecessor.begin(), examinedPredecessor.end(), suffix);
293  examinedPredecessor.erase(kt);
294  return equivalent(i, j, examinedPredecessor);
295  });
296  while (jt != allPredecessors.end()) {
297  allPredecessors.erase(jt);
298  if (allPredecessors.empty()) {
299  break;
300  }
301  jt = std::find_if_not(allPredecessors.begin(), allPredecessors.end(), [&](const auto &suffix) {
302  auto examinedPredecessor = allPredecessors;
303  auto jt = std::find(examinedPredecessor.begin(), examinedPredecessor.end(), suffix);
304  examinedPredecessor.erase(jt);
305  return equivalent(i, j, examinedPredecessor);
306  });
307  }
308 
309  suffixes.reserve(suffixes.size() + allPredecessors.size());
310  std::move(allPredecessors.begin(), allPredecessors.end(), std::back_inserter(suffixes));
311  }
312  this->refreshTable();
313 
314  return false;
315  }
316  }
317  }
318  }
319  return true;
