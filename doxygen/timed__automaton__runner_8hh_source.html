72  if (isEmpty) {
73  return false;
74  }
75  assert(this->clockValuation.size() == this->automaton.maxConstraints.size());
76  // Skip the operation if we are at the sink state
77  if (this->state == nullptr) {
78  return false;
79  }
80  for (const TATransition &transition: this->state->next[action]) {
81  // Check if the guard is satisfied
82  if (std::all_of(transition.guard.begin(), transition.guard.end(), [&](const Constraint &guard) {
83  return guard.satisfy(this->clockValuation.at(guard.x));
84  })) {

98  bool step(double duration) override {
99  if (isEmpty) {
100  return false;
101  }
102  assert(this->clockValuation.size() == this->automaton.maxConstraints.size());
103  // Skip the operation if we are at the sink state
104  if (this->state == nullptr) {
105  return false;
106  }
107  if (duration == 0) {
108  return this->state->isMatch;
109  }
110  const auto unobservableIt = this->state->next.find(UNOBSERVABLE);
111  if (unobservableIt != this->state->next.end() && !unobservableIt->second.empty()) {
112  // If there are unobservable transitions, we choose the minimum duration to satisfy it
113  Bounds minDuration;
114  auto candidateTransition = unobservableIt->second.end();
115  if (unobservableIt->second.size() == 1) {
116  minDuration = lowerBoundDurationToSatisfy(unobservableIt->second.front().guard, this->clockValuation);
117  candidateTransition = unobservableIt->second.begin();
118  } else {
119  candidateTransition = std::min_element(unobservableIt->second.begin(), unobservableIt->second.end(),
120  [&](const auto &a, const auto &b) {
121  return lowerBoundDurationToSatisfy(a.guard, this->clockValuation) <
122  lowerBoundDurationToSatisfy(b.guard, this->clockValuation);
123  });
124  minDuration = lowerBoundDurationToSatisfy(candidateTransition->guard, this->clockValuation);
125  }
126  if (std::isfinite(minDuration.first) && minDuration.first <= 0 && Bounds{-duration, true} <= minDuration) {
127  // An unobservable transition is available
128  std::transform(this->clockValuation.begin(), this->clockValuation.end(), this->clockValuation.begin(),
129  [&](double value) {
130  return value + -minDuration.first;
131  });
132  // Reset the clock variables
133  // assert the validity of the candidate transition
134  assert(candidateTransition != unobservableIt->second.end());
135  this->applyReset(candidateTransition->resetVars);
136  this->state = candidateTransition->target;
137  return this->step(duration - -minDuration.first);
138  } else {
139  BOOST_LOG_TRIVIAL(debug) << "Unobservable transitions are skipped";
140  BOOST_LOG_TRIVIAL(debug) << "std::isfinite(-minDuration): " << std::isfinite(-minDuration.first);
141  BOOST_LOG_TRIVIAL(debug) << "minDuration: " << minDuration.first << ", " << minDuration.second;
142  BOOST_LOG_TRIVIAL(debug) << "duration: " << duration;
143  }
144  }
145  // No unobservable transition is available
146  std::transform(this->clockValuation.begin(), this->clockValuation.end(), this->clockValuation.begin(),
147  [&](double value) {
148  return value + duration;
149  });
150 
151  return this->state->isMatch;
152  }
