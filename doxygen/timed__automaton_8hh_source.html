162  void removeDeadLoop() {
163  std::deque<std::shared_ptr<TAState>> nonAccepting;
164  std::copy_if(this->states.begin(), this->states.end(), std::back_inserter(nonAccepting),
165  [](const std::shared_ptr<TAState>& state) {
166  return !state->isMatch;
167  });
168  while (!nonAccepting.empty()) {
169  std::shared_ptr<TAState> current = nonAccepting.front();
170  nonAccepting.pop_front();
171  // We do not remove the initial states
172  if (std::find(this->initialStates.begin(), this->initialStates.end(), current) != this->initialStates.end()) {
173  continue;
174  }
175  if (std::all_of(current->next.begin(), current->next.end(), [&](const auto &pair) {
176  return std::all_of(pair.second.begin(), pair.second.end(), [&] (const auto &transition) {
177  return transition.target == current.get();
178  });
179  })) {
180  this->states.erase(std::remove(this->states.begin(), this->states.end(), current), this->states.end());
181  // Remove the transition to the removed location
182  for (auto &state: this->states) {
183  for (auto it = state->next.begin(); it != state->next.end(); ) {
184  auto &[action, transitions] = *it;
185  transitions.erase(std::remove_if(transitions.begin(), transitions.end(), [&] (const auto &transition){
186  return transition.target == current.get();

188
