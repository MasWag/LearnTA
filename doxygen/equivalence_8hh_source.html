103  static inline RenamingGraph toGraph(const TimedCondition &left, const TimedCondition &right) {
104  // Assert the precondition
105  assert(left.isSimple());
106  assert(right.isSimple());
107  std::vector<std::vector<std::size_t>> v1Edges, v2Edges;
108 
109  const auto N = left.size();
110  const auto M = right.size();
111  v1Edges.resize(N);
112  v2Edges.resize(M);
113  std::size_t v1 = 0, v2 = 0;
114  std::vector<std::size_t> currentSameV1, currentSameV2;
115  while (v1 < v1Edges.size() && v2 < v2Edges.size()) {
116  if (left.getUpperBound(v1, N - 1) == right.getUpperBound(v2, M - 1)) {
117  v1Edges.reserve(currentSameV2.size());
118  v2Edges.reserve(currentSameV1.size());
119  std::copy(currentSameV2.begin(), currentSameV2.end(), std::back_inserter(v1Edges.at(v1)));
120  std::for_each(currentSameV2.begin(), currentSameV2.end(), [&](const auto oldV2) {
121  v2Edges.at(oldV2).push_back(v1);
122  });
123  std::copy(currentSameV1.begin(), currentSameV1.end(), std::back_inserter(v2Edges.at(v2)));
124  std::for_each(currentSameV1.begin(), currentSameV1.end(), [&](const auto oldV1) {
125  v1Edges.at(oldV1).push_back(v2);
126  });
127  v1Edges.at(v1).push_back(v2);
128  v2Edges.at(v2).push_back(v1);
129  currentSameV1.push_back(v1);
130  currentSameV2.push_back(v2);
131  if (v1 + 1 < v1Edges.size() && left.getUpperBound(v1, N - 1) == left.getUpperBound(v1 + 1, N - 1)) {
132  v1++;
133  } else {
134  v2++;
135  }
136  } else {
137  currentSameV1.clear();
138  currentSameV2.clear();
139  if (left.getUpperBound(v1, N - 1) < right.getUpperBound(v2, M - 1)) {
140  v2++;
141  } else {
142  v1++;
143  }
144  }
145  }
146 
147  // Sort and remove duplication
148  for (auto& v1Edge: v1Edges) {
149  std::sort(v1Edge.begin(), v1Edge.end());
150  v1Edge.erase(std::unique(v1Edge.begin(), v1Edge.end()), v1Edge.end());

165  static inline std::vector<RenamingRelation> generateDeterministicCandidates(const TimedCondition& left,
166  const TimedCondition& right,
167  const RenamingGraph& graph) {
168  std::vector<RenamingRelation> candidates;
169  // We first generate full candidates
170  candidates.emplace_back();
171  for (std::size_t j = 0; j < right.size(); ++j) {
172  // We do not construct a renaming equation if there is no edge from the right node, or the target is already determined
173  if (graph.second.at(j).empty() || right.getUpperBound(j, right.size() - 1).second) {
174  continue;
175  }
176  std::vector<RenamingRelation> newCandidates;
177  for (auto candidate: candidates) {
178  // If j - 1 and j has the same value, we use the same left variable
179  if (j > 0 && right.getUpperBound(j - 1, j - 1) == Bounds{0, true}) {
180  candidate.emplace_back(candidate.back().first, j);
181  newCandidates.emplace_back(std::move(candidate));
182  continue;
183  } else {
184  // The least value of the corresponding node
185  const std::size_t lowerBound = candidate.empty() ? 0 : (candidate.back().first + 1);
186  for (const auto& i: graph.second.at(j)) {
187  // We skip if i - 1 and i has the same value
188  if (i >= lowerBound && (i == 0 || left.getUpperBound(i - 1, i - 1) != Bounds{0, true})) {
189  auto tmpCandidate = candidate;
190  tmpCandidate.emplace_back(i, j);
191  newCandidates.emplace_back(std::move(tmpCandidate));
192  }
193  }
194  }
195  }
196  candidates = std::move(newCandidates);
197  }
198 
199  // Then, we add empty renaming
200  candidates.emplace_back();
201  // Finally, erase non-deterministic renaming
202  candidates.erase(std::remove_if(candidates.begin(), candidates.end(), [&] (const auto& renaming) {
203  return renaming.hasImpreciseClocks(right);
204  }), candidates.end());
205  std::sort(candidates.begin(), candidates.end());
206  candidates.erase(std::unique(candidates.begin(), candidates.end()), candidates.end());
207 
208  return candidates;
209  }

335  leftConcatenations.resize(leftRow.size());
336  rightConcatenations.resize(leftRow.size());

338  std::vector<std::size_t> constrainedV1, constrainedV2;

343  if (leftRow.at(i).empty() != rightRow.at(i).empty()) {
344  // One of them is bottom but another is not
345  return std::nullopt;
346  }

348  leftRow.at(i).getStrictlyConstrainedVariables(leftConcatenations.at(i), N);

369  candidates.emplace_back();

372  while (v1Index < constrainedV1.size() && v2Index < constrainedV2.size()) {
373  std::size_t v1 = constrainedV1.at(v1Index);
374  std::size_t v2 = constrainedV2.at(v2Index);
375  // 3-1. We check if we cannot include these vertices
376  if (v1Edges.at(v1).empty()) {

411  tmp.emplace_back();

421  v1Index++;
422  }
423  while (v2Index < constrainedV2.size() && constrainedV2.at(v2Index) <= v1Edges.at(v1).back()) {
424  v2Index++;
425  }
426  }
427  }
