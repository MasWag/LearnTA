54  std::make_move_iterator(elementaryLanguages.end()),
55  timedConditionsWithSize.begin(),
56  [&](auto &&elementary) {
57  return std::make_pair(elementary.getTimedCondition(), 1);
58  });
59  auto it = timedConditionsWithSize.begin();
60  while (it != timedConditionsWithSize.end()) {

63  for (auto it2 = std::next(it); it2 != timedConditionsWithSize.end(); it2++) {
64  // Check if the convex hull is the exact union

66  std::vector<TimedCondition> enumerated;

69  it->first = std::move(convexHull);

72  it = timedConditionsWithSize.begin();

74  break;
