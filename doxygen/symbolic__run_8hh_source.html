59  [[nodiscard]] TimedWord reconstructWord() const {
60  // Sample the last concrete state;
61  auto postZoneState = this->back();
62  auto postZone = postZoneState->zone;
63  auto postValuation = postZone.sample();
64 
65  const auto N = this->edges.size();
66  std::list<double> durations;
67 
68  for (int i = N - 1; i >= 0; --i) {
69  auto preZoneState = this->stateAt(i);
70  auto preZone = preZoneState->zone;
71 
72  // Construct the zone just before jump
73  auto zoneBeforeJump = Zone{postValuation, postZone.M};
74  if (!zoneBeforeJump.isSatisfiableNoCanonize()) {
75  BOOST_LOG_TRIVIAL(error) << "Failed to reconstruct word from a symbolic run\n" << *this;
76  }
77  assert(zoneBeforeJump.isSatisfiableNoCanonize());
78  const auto transition = this->edgeAt(i);
79  for (const auto &[resetVariable, updatedVariable]: transition.resetVars) {
80  if (!updatedVariable) {
81  zoneBeforeJump.unconstrain(resetVariable);
82  }
83  }
84  const auto originalZoneBeforeJump = zoneBeforeJump;
85  for (const auto &[resetVariable, updatedVariable]: transition.resetVars) {
86  if (updatedVariable) {
87  zoneBeforeJump.value.col(*updatedVariable + 1) = originalZoneBeforeJump.value.col(resetVariable + 1);
88  zoneBeforeJump.value.row(*updatedVariable + 1) = originalZoneBeforeJump.value.row(resetVariable + 1);
89  zoneBeforeJump.unconstrain(resetVariable);
90  }
91  }
92  assert(zoneBeforeJump.isSatisfiableNoCanonize());
93  for (const auto guard: transition.guard) {
94  zoneBeforeJump.tighten(guard);
95  if (!zoneBeforeJump.isSatisfiableNoCanonize()) {
96  BOOST_LOG_TRIVIAL(error) << "Failed to reconstruct word from a symbolic run\n" << *this;
97  BOOST_LOG_TRIVIAL(error) << "The unsatisfiable zone\n" << zoneBeforeJump;
98  }
99  assert(zoneBeforeJump.isSatisfiableNoCanonize());
100  }
101  {
102  auto tmpPreZone = preZone;
103  tmpPreZone.elapse();
104  zoneBeforeJump &= tmpPreZone;
105  assert(zoneBeforeJump.isSatisfiableNoCanonize());
106  }
107 
108  // Sample the valuation just before discrete jump
109  assert(zoneBeforeJump.isSatisfiableNoCanonize());
110  const auto valuationBeforeJump = zoneBeforeJump.sample();
111  auto backwardPreZone = Zone{valuationBeforeJump, postZone.M};
112  backwardPreZone.reverseElapse();
113  const auto preValuation = (preZone && backwardPreZone).sample();
114  durations.push_front(valuationBeforeJump.at(0) - preValuation.at(0));
115 
116  // Update the variables
117  postZone = std::move(preZone);
118  postValuation = preValuation;
119  }
120  // Return the timed word
121  std::vector<double> durationsVector;
122  durationsVector.resize(durations.size());
123  std::move(durations.begin(), durations.end(), durationsVector.begin());
124  durationsVector.push_back(0);
125 
126  return {word, durationsVector};
127  }
